# Optional: Public Liquidity Pool Configuration
# -----------------------------------------------------------------------------
#
# Pools are configured using the pattern:
#   VITE_POOL_<POOL_KEY>_ADDRESS=0x...
#   VITE_POOL_<POOL_KEY>_TOKEN_ADDRESS=0x...      (optional)
#   VITE_POOL_<POOL_KEY>_TOKEN_SYMBOL=AGS/S   (optional display label)
#   VITE_POOL_<POOL_KEY>_LP_SYMBOL=AGS-LP         (optional display label)
#   VITE_POOL_<POOL_KEY>_USE_NATIVE=true|false
#
# Example:
VITE_POOL_WETH_ADDRESS=0xD7c29aC1E56d2E5720A30890F491E54AE2e5A5aE
VITE_POOL_WETH_TOKEN_SYMBOL=WETH
VITE_POOL_WETH_TOKEN_ADDRESS=0x4200000000000000000000000000000000000006
VITE_POOL_WETH_USE_NATIVE=true
VITE_POOL_WETH_LP_SYMBOL=AGS-WETH-LP
